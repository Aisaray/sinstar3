输入之星皮肤格式

    输入之星提供了灵活的换肤功能，你不仅可以在输入之星中选择不同的背景图案，甚至您也可以控制输入之星各显示单元的显示位置。 输入之星皮肤保存在输入之星皮肤目录（安装目录\skins)，包含一个配置文件config.ini和相关的图片。您可以使用启程软件提供的“输入之星皮肤编辑器”来制作皮肤（官方网站下载），也可以手工编辑配置文件。

配置文件示例(config.ini)：

;图片格式说明：图片可以是bmp、png，png文件必须是32位即包含一个透明通道
;如果是png格式，应该在前而加一个描述项：imgtype=1,系统默认状态为位图即imgtype=0.
[desc]
name=green2
author=与狼共舞

[InputWindow]
;输入窗口配置
imgtype=0
image=skin1.bmp
color_trans=0,0,204
;背景图
left=40
right=160
;左右边宽度,由于窗口宽度根据显示的内容而改变，将图片分成左，中，右三部分，中间部分平铺
font_name=宋休
font_height=16
;字体信息
margin=27,48,110,12
;定义边界left,top,right,bottom
heteromorphic=1
;指示窗口是否使用不规则窗口
margin_comp=0,0
;编码区相对于左右边界的附加边界
margin_cand=10,0
;重码区相对于左右边界的附加边界
charmode=charmode.png
;在输入窗口上显示标点状态使用的图，必须为32位的PNG图，水平方向上包含2个子图，左边为中文标点，右边为英文标点。支持半透明。


[Color_Comp]
;编码窗口的颜色配置
normal=0,0,0

;当前编辑音节的拼音

;当前编辑的音节

;当前编辑的音节的下划线

;自定义编码

userdef_separate=0,0,255

;一般重码

;联想得到的编码

;万能键所在位置的编码
en_head=128,0,0
;英文头
en_phonetic=0,0,255
;英文音标
tip=0,0,0
;编码提示信息
